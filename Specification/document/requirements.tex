\subsection{Nonfunctional Requirements}

\nonFunctionality{Available configuration options have a concise description}{nfc:intuitiv}

\nonFunctionality{Storage of benchmark results is persistent}{nfc:persistent}

\nonFunctionality{Generation of plots does not take more than 5 seconds}{nfc:runtime}

\nonFunctionality{System can handle malformed data and invalid requests without crashing and provides a descriptive error message}{nfc:robust}
